		$_haveImg=$(this).parents('.yuyue_vertop').find('.content_jumpDiv .have_ewm img');
		if($_haveImg){
			$_haveImg.attr('src',$_haveImg.attr('_imgsrc'));
		}
		$('.yuyue_vertop').css('z-index',99);
		$(this).parents('.yuyue_vertop').css('z-index',999);
		var tel = $(this).parents('.yuyue_vertop').find('.guishudi').attr("tel");
		var tel_city= $(this).parents('.yuyue_vertop').find('.guishudi');
		$.ajax({
            type: "GET",
            url:"http://huangye.api.58.com/phonearea/?phone="+tel,
            dataType: "jsonp",
            jsonp:"callback",
            callback:"aaa",
            success: function(data) {
				var phonearea = data.area;
				if(phonearea!=null && phonearea!=""){
					tel_city.html("归属地("+data.area+")");
				}                
            }
        })
		$('.content_jumpDiv').fadeOut('fast');
		$(this).parents('.yuyue_vertop').find('.content_jumpDiv').fadeIn("slow");		
	})
	$('.yuyue_vertop .close_btn').click(function(){
		$('.yuyue_vertop').css('z-index',99);
		$('.content_jumpDiv').fadeOut('slow')
	})
})
