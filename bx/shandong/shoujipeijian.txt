		var colors = ['#5ece76','#50b1e3','#f88788'];

			url: ad.curl,

			line2: ad.desc,
			visible_url: ad.surl,
			color: randomColor()
		},template);
	}
	function initRender(_ads) {
		ads = []

					ads.push(_ads[i])
				}
			}

		if (loadedAdSize) {
			colsizes = $('.waterfall-glist').map(function(){
				return $(this).find('.waterdrop').length;
			});
		} else {
			colsizes = [-1,0,0,0,0];
			loadedAdSize = 20;
		}
		setTimeout(function(){
			var insertSize = Math.ceil(loadedAdSize/10) + 1;
			for(var i = 0; i < insertSize; i++ ) {
				var col = Math.floor(Math.random() * colsizes.length);

				insertNode.before( renderAd() );
			}
		}, loadedAdSize*100);
	}
	return {

		render: render
	};
