		if(clickedAdsense.indexOf(id) === -1) {
			clickedAdsense.push(id)
		} else {
			setTimeout(function() {
				$(id).remove()
			}, 1000)
		}
	})

	// 暂时检测一下搜索广告点击状况，就不取全量了
	if (Math.random() < 0.1) {
		$('body').on('click', '.media.clearfix.adsense', function(e) {
			var offset = $(e.target).parents('.adsense').offset()
			var posY = offset.top - $(window).scrollTop()
			var posX = offset.left - $(window).scrollLeft()

			var result = {
				x: e.screenX - posX,
				y: e.screenY - posY - 120
			}

			tracker.evt('search_ad_click_position', result)
		})
	}
})();
